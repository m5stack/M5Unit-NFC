58constexpr inline TagBits tag_to_tagbit(const Tag t)
59{
60 return (t == Tag::Null) ? (1u << 0)
61 : (t == Tag::LockControl) ? (1u << 1)
62 : (t == Tag::MemoryControl) ? (1u << 2)
63 : (t == Tag::Message) ? (1u << 3)
64 : (t == Tag::Proprietary) ? (1u << 4)
65 : (t == Tag::Terminator) ? (1u << 5)
66 : 0u;
67}
